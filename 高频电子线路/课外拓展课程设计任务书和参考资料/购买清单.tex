	\section*{\textcolor{red}{[重要]} 采购前必读（给买手队友）}
	\begin{enumerate}
		\item \textbf{STM32 开发板：} \textcolor{blue}{不需要买}（组长已买好）。
		\item \textbf{贴片封装统一：} 所有电阻、电容、电感，除非特别标注，**统统买 0603 封装**。
		\item \textbf{L3 的陷阱：} 原理图和 BOM 表里的 \textbf{L3} 其实是 \textbf{15pF 的电容}！千万别买成电感！
		\item \textbf{数量策略：} 小元件（电容电感）很便宜但容易丢，**请按清单里的“推荐购买量”下单**（通常是 10-20 个起），多买点备用。
		\item \textbf{电感品牌：} 不需要买昂贵的进口货，搜**“0603 叠层电感”**，买国产普通品牌即可。
	\end{enumerate}
